## 1. Problem Statement

Playground is the only place admin can answer "what actually happens when employee X sends message Y?" But currently:
- Simulate mode uses keyword matching + canned responses (not real AI)
- `_POS_TOOLS` hardcoded dict duplicates DynamoDB POS#.toolAllowlist
- Pipeline tab shows static config, not runtime events from the actual interaction
- Files tab references SOUL.md instead of PERSONAL_SOUL.md
- No visibility into AUDIT# events generated by the test interaction
- No KB, guardrail, or model information in pipeline view

### Design Intent

```
Playground = Interactive Audit Replay

Left:  Conversation with real AgentCore (Live) or Bedrock Converse (Simulate)
Right: Real-time pipeline showing what ACTUALLY happened during this interaction
       - Configuration summary (loaded once when employee selected)
       - Interaction events (refreshed after each message from AUDIT#)
```

### 2.2 Pipeline API — static config + dynamic events

**New endpoint: configuration summary**
```python
@router.get("/api/v1/playground/pipeline/{emp_id}")
def get_pipeline_config(emp_id: str, authorization: str = Header(default="")):
    """Get complete runtime configuration for an employee's agent."""
    require_role(authorization, roles=["admin", "manager"])
    emp = db.get_employee(emp_id)
    if not emp:
        raise HTTPException(404)
    pos_id = emp.get("positionId", "")
    pos = db.get_position(pos_id) if pos_id else {}

    # SOUL summary
    global_soul = s3ops.read_file("_shared/soul/global/SOUL.md") or ""
    position_soul = s3ops.read_file(f"_shared/soul/positions/{pos_id}/SOUL.md") if pos_id else ""
    personal_soul = s3ops.read_file(f"{emp_id}/workspace/PERSONAL_SOUL.md") or ""

    # Model resolution
    from routers.settings import _get_model_config
    mc = _get_model_config()
    model = (mc.get("employeeOverrides", {}).get(emp_id, {}).get("modelId")
             or mc.get("positionOverrides", {}).get(pos_id, {}).get("modelId")
             or mc.get("default", {}).get("modelId", ""))

    # KB assignments
    kb_cfg = db.get_config("kb-assignments") or {}
    kb_ids = list(set(kb_cfg.get("positionKBs", {}).get(pos_id, [])
                      + kb_cfg.get("employeeKBs", {}).get(emp_id, [])))

    # Runtime
    routing = db.get_routing_config()
    runtime_id = (routing.get("employee_override", {}).get(emp_id)
                  or routing.get("position_runtime", {}).get(pos_id)
                  or "default")

    return {
        "employee": {"id": emp_id, "name": emp.get("name",""), "position": pos.get("name","")},
        "soul": {
            "globalWords": len(global_soul.split()),
            "positionWords": len(position_soul.split()),
            "personalWords": len(personal_soul.split()),
        },
        "planA": {
            "tools": pos.get("toolAllowlist", ["web_search"]),
            "blocked": [t for t in ["shell","browser","file_write","code_execution"]
                        if t not in pos.get("toolAllowlist", ["web_search"])],
        },
        "kbs": kb_ids,
        "model": model,
        "runtime": runtime_id,
    }
```

**New endpoint: interaction events**
```python
@router.get("/api/v1/playground/events")
def get_playground_events(tenant_id: str = "", seconds: int = 60, authorization: str = Header(default="")):
    """Get AUDIT# events for a specific tenant from the last N seconds."""
    require_role(authorization, roles=["admin", "manager"])
    entries = db.get_audit_entries(limit=50)
    from datetime import datetime, timezone, timedelta
    cutoff = (datetime.now(timezone.utc) - timedelta(seconds=seconds)).isoformat()
    # Match tenant_id in actorId (may be emp-xxx or the full pgnd__emp-xxx__hash)
    base_id = tenant_id.replace("port__", "").replace("pgnd__", "").split("__")[0] if tenant_id else ""
    events = [e for e in entries
              if e.get("timestamp", "") >= cutoff
              and (base_id in e.get("actorId", "") or base_id in e.get("targetId", ""))]
    return {"events": events, "count": len(events)}
```

### 2.3 Simulate mode — Bedrock Converse with real SOUL

Replace keyword matching with actual Bedrock call using the employee's real configuration:

```python
# Simulate mode: use Bedrock Converse with employee's actual SOUL + Plan A
# Faster than Live (no AgentCore cold start), but uses real SOUL and permissions

import boto3 as _b3_sim

def _simulate_agent(emp_id: str, message: str, profile: dict) -> dict:
    """Simulate agent response using Bedrock Converse with real SOUL."""
    pos_id = next((e.get("positionId","") for e in db.get_employees() if e["id"] == emp_id), "")

    # Build system prompt from real SOUL layers
    global_soul = s3ops.read_file("_shared/soul/global/SOUL.md") or ""
    position_soul = s3ops.read_file(f"_shared/soul/positions/{pos_id}/SOUL.md") if pos_id else ""
    personal_soul = s3ops.read_file(f"{emp_id}/workspace/PERSONAL_SOUL.md") or ""

    system = f"{global_soul}\n\n---\n\n{position_soul}\n\n---\n\n{personal_soul}"

    # Inject Plan A
    tools = profile.get("tools", [])
    blocked = [t for t in ["shell","browser","file_write","code_execution"] if t not in tools]
    if blocked:
        plan_a = f"Allowed tools: {', '.join(tools)}.\nYou MUST NOT use: {', '.join(blocked)}."
        system = f"{plan_a}\n\n---\n\n{system}"

    try:
        from shared import GATEWAY_REGION
        bedrock = _b3_sim.client("bedrock-runtime", region_name=GATEWAY_REGION)
        model_id = os.environ.get("BEDROCK_MODEL_ID", "global.amazon.nova-2-lite-v1:0")
        resp = bedrock.converse(
            modelId=model_id,
            system=[{"text": system[:8000]}],  # cap system prompt
            messages=[{"role": "user", "content": [{"text": message}]}],
            inferenceConfig={"maxTokens": 2048},
        )
        reply = resp["output"]["message"]["content"][0]["text"]
        return {"response": reply, "source": "simulate-bedrock"}
    except Exception as e:
        return {"response": f"Simulation error: {e}", "source": "error"}
```
